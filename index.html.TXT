<script>
    function calculateTotal() {
        var price = 2500; // Price fix hai
        var qty = document.getElementById('pQty').value;
        document.getElementById('pTotal').innerText = price * qty;
    }

    function addToCart() {
        var productData = {
            title: "Luxury Wrist Watch",
            price: 2500,
            qty: document.getElementById('pQty').value,
            total: document.getElementById('pTotal').innerText
        };
        // Data browser mein save karein
        localStorage.setItem('myOrder', JSON.stringify(productData));
        
        // Checkout page par bhejein
        window.location.href = "checkout.html"; 
    }
</script>

    <thead>
        <tr>
            <th>Date</th>
            <th>Name</th>

<script>
    var orders = JSON.parse(localStorage.getItem('sellerOrders')) || [];
    var tbody = document.querySelector("#orderTable tbody");

    if(orders.length === 0) {
        tbody.innerHTML = "<tr><td colspan='6'>No orders yet.</td></tr>";
    } else {
        // Naye orders pehle dikhaye
        orders.reverse().forEach(function(o) {
            var row = `<tr>
                <td>${o.date}</td>
                <td>${o.name}</td>
                <td>${o.mobile}</td>
                <td>${o.item} (x${o.qty})</td>
                <td>Rs. ${o.total}</td>
                <td>${o.address}</td>
            </tr>`;
            tbody.innerHTML += row;
        });
    }

    function clearAll() {
        if(confirm("Delete all orders?")) {
            localStorage.removeItem('sellerOrders');
            location.reload();
        }
    }
</script>
